Flights2_incomplete\(Expected_Time<-Flights2_incomplete\)Distance*0.01 ## Expected vliegtijd berekening (oranje) Flights2_incomplete\(Res<-Flights2_incomplete\)Flighttime_min - Flights2_incomplete$Expected_Time # Residuals berekening van oranje rechte

modelOrange<-sum(Flights2_incomplete\(Res^2, na.rm= TRUE) ## squared residuals (oranje) modelGreen<-sum(model_dist_all\)residuals^2) ## squared residuals (groen)

dfOrange<-length(na.omit(Flights2_incomplete\(Distance))-2 # degrees of freedom berekening (obs-interecept -slope) dfGreen<-length(na.omit(Flights2_incomplete\)Flighttime_min))-2 # degrees of freedom berekening (obs-int. -slope)

Fstat<-((modelOrange-modelGreen)/(dfOrange-dfGreen))/(modelGreen/dfGreen) # F statistic pf(Fstat, dfOrange, dfGreen, lower.tail = FALSE)
